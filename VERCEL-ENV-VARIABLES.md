# Vercel Environment Variables
# Vercel Dashboard → Your Project → Settings → Environment Variables

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
VARIABLE — Groq API Key (REQUIRED)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
Name:   GROQ_API_KEY
Value:  [paste your key from console.groq.com/keys]

Check:  ✅ Production  ✅ Preview  ✅ Development

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
STEPS:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
1. vercel.com/dashboard → click writeclean project
2. Settings → Environment Variables → Add New
3. Name = GROQ_API_KEY
4. Value = your key from console.groq.com/keys
5. Check all 3 environments → Save
6. Deployments tab → 3 dots on latest → Redeploy
7. Test: paste any text → click Check Grammar ✓

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
